$("#load-it").on('click', function(){
  var root = 'https://jsonplaceholder.typicode.com';
  $.ajax({
    url: root + '/posts/1',
    method: 'GET'
  }).then(function(data) {
    $("#change-it #title").text(data.title)
  });
})
